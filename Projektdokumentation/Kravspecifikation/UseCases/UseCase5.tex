\subsection{Use case 5}
I denne use case opretter tekniker et nyt kar. 
\begin{usecase}

\addtitle{Use Case 5}{Opret kar} 

\additemizedfield{Mål:}{
\item At oprette et nyt kar i systemet
}

\addfield{Initieret af:}{
	Tekniker   
}

\additemizedfield{Aktører:}{\item Primær: Tekniker}

\addfield{Samtidige forekomster:}{1}

%Preconditions: What must be true on start and worth telling the reader?
\addfield{Prækondition:}{Der er tilkoblet et kar til systemet}

%Postconditions: What must be true on successful completion and worth telling the reader
\addfield{Postkondition:}{Der er oprettet et kar }

%Main Success Scenario: A typical, unconditional happy path scenario of success.
\addscenario{Hovedscenarie:}{
	\item Tekniker trykker på "Opret kar" på GUI 
	\begin{itemize}
	\item Systemet viser en menu hvor det er muligt at
	      indtaste volumen og adresse i et nyt kar 
	\end{itemize}	 		
	\item Tekniker trykker på "Volumen"
	\begin{itemize}
	\item Systemet viser en cursor i skrivefeltet tilhørende
		  volumen
	\end{itemize}
	\item Tekniker indtaster en volumen i enheden Liter
	\begin{itemize}
	\item Systemet viser den nye værdi 
	\end{itemize}	
	\item Tekniker trykker på "Adresse"
	\begin{itemize}
	\item Systemet viser en cursor i skrivefeltet tilhørende\\
	      adressen
	\end{itemize}
	\item Tekniker indtaster adressen på karet 
	\begin{itemize}
	\item Systemet viser den nye værdi
	\end{itemize}	 
	\item Tekniker trykker på "Tilføj kar"
	\begin{itemize}
	\item Systemet opretter det nye kar og retunerer til hoved
	      menuen
	\end{itemize}
	\begin{itemize}
	\item Det nye kar forekommer nu i hoved menuen
	\end{itemize}
}

\end{usecase}
